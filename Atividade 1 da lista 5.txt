let peso = parseFloat(prompt("Digite seu peso (em kg): "));
let altura = parseFloat(prompt("Digite sua altura (em metros): "));

let imc = peso / (altura * altura);

alert("Seu IMC Ã©: " + imc.toFixed(2));

if (imc < 18.5) {
  alert("Abaixo do peso");
} else if (imc < 25) {
  alert("Peso normal");
} else if (imc < 30) {
  alert("Acima do peso");
} else {
  alert("Obesidade");
}
